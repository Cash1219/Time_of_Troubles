weakness = { #弱能力者
	texture = "gfx/interface/icons/pops_icons/weakness.dds"
	color = hsv{ 0.02 0.38 0.37 }
	unemployment = yes
	unemployment_wealth = 15

	can_always_hire = yes
	start_quality_of_life = 5
	wage_weight = 2
	paid_private_wage = yes
	dependent_wage = 0.5	# per year

	# 10% - 70% politically engaged
	political_engagement_base = 0.1
	political_engagement_literacy_factor = 0.7

	political_engagement_mult = {
		value = 1.0

		add = {
			desc = "POP_STARVATION"

			if = {
				limit = {
					is_in_mild_starvation = yes
				}
				value = 0.25
			}
			else_if = {
				limit = {
					is_in_severe_starvation = yes
				}
				value = 1
			}
		}

		multiply = {
			desc = "NO_LABOR_MOVEMENT"
			value = 1.0

			if = {
				limit = {
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.75
			}
		}

		multiply = {
			desc = "HAS_POLITICAL_AGITATION"
			value = 1.0

			if = {
				limit = {
					owner = {
						has_technology_researched = political_agitation
					}
				}
				value = 1.5
			}
		}
	}
	qualifications_growth_desc = "weakness_QUALIFICATIONS_DESC"
	qualifications = {
			limit = {
				if = {
					state = {
						has_law = law_type:law_press_education
					}
				}
				literacy_rate > 0.2
				else = {
					literacy_rate > 0.4
				}
			}
			add = {
			value = wealth
			subtract = 10
			min = 0
			desc = "QUALIFICATIONS_WEALTH_FACTOR"
		}
		if = {
			limit = {
				state = {
				has_law = law_type:law_press_education
				}
			}
			add = {
				value = literacy_rate
				subtract = 0.2
				multiply = 30
				desc = "QUALIFICATIONS_PRESS_EDUCATION_FACTOR"
			}
		}

		if = {
			limit = {
				state = {
					has_law = law_type:law_moneycontrol_rights
				}
			}
			add = {
				value = wealth
				subtract = 10
				min = 0
				multiply = 3
				desc = "QUALIFICATIONS_WEALTH_FACTOR"
			}
		}

		if = {
			limit = {
				state = {
					has_law = law_type:law_familycontrol_rights
				}
			}
			if = {
				is_pop_type = aristocrats
			}
			multiply = 10
			else = {
				multiply = 0.1
			}
		}

		if = {#神授体系下异教徒
			limit = {
				state = {
					state_religious_acceptance = {
							target = ROOT.religion
							value < religious_acceptance_medium
					   	}
					has_law = law_type:law_godbless_rights
				}
				multiply = {
					value = 0
					desc = "POTENTIALS_RELIGIOUS_DISCRIMINATION"
				}
			}
		}
			if = {#神授体系下异教徒
			limit = {
				state = {
					has_law = law_type:law_godbless_rights
				}
				is_pop_type = clergymen
			}
				add = {
					value = 10
					desc = "POTENTIALS_RELIGIOUS_DISCRIMINATION"
				}
		}
		
	}
	portrait_age = {
		integer_range = {
			min = define:NPortrait|GRACEFUL_AGING_START
			max = define:NPortrait|GRACEFUL_AGING_END
		}
	}
	portrait_pose = { value = 0 }
	portrait_is_female = { always = yes }

}
