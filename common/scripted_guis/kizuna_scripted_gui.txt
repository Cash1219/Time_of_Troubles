kon_kizuna_list = {
	effect = {
		if = {
			limit = {
				var:kon_kizuna_number < 2
				remove_modifier = kizuna_common_modifier_1
				remove_modifier = kizuna_common_modifier_2
				remove_modifier = kizuna_common_modifier_3
				remove_modifier = kizuna_common_modifier_4
				remove_modifier = kizuna_common_modifier_5
				remove_modifier = kizuna_common_modifier_6
			}
		}
	}
	is_shown = {
		any_scope_character = {
			OR = {
				has_trait = tot_Kon_Yui_trait
				has_trait = tot_Kon_Mio_trait
				has_trait = tot_Kon_Tsumugi_trait
				has_trait = tot_Kon_Ritsu_trait
				has_trait = tot_Kon_Azusa_trait
			}
		}
    } 		
    
	is_valid = {
		
	}
}
