fra.2 = { #面粉暴动
	type = country_event
	placement = root
	title = fra.2.t
	desc = fra.2.d
	flavor = fra.2.f

	duration = 3

	event_image = {
		video = "unspecific_factory_closed"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	
	trigger = {
		country = c:FRA
		OR = {
			mg:grain = {
				market_goods_cheaper > 0.3
			}
			root.average_sol < 10
		}
		NOT = {
			has_variable = event_fra_2_var
		}
	}
	immediate = {
		set_variable = event_fra_2_var
		random_scope_building = {
			limit = {
				is_building_group = bg_wheat_farms 
				NOT = {
					has_modifier = modifier_fra_flour_war
				}
			}
			add_modifier = {
				name = modifier_fra_flour_war
				months = 12
			}
		}
	}
	option = {
		name = fra.2.a #救
		add_treasury = {
			value = gdp
			multiply = -0.05
		}
		add_modifier = {
			name = modifier_fra_emergency_supply
			years = 10
		}
	}
	option = {
		name = fra.2.b #演
		add_modifier = {
			name = modifier_fra_pretend_conceal
			years = 10
		}
		add_radicals = {
			value = medium_radicals
			strata = poor
		}
	}
	option = {
		name = fra.2.c #摆
		add_radicals = {
			value = large_radicals
			strata = poor
		}
	}
}
